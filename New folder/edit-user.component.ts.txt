import { Component } from '@angular/core';
import service{=CrudService} from '../crud.service';
@Component({
  selector: 'app-edit-user',
  templateUrl: './edit-user.component.html',
  styleUrls: ['./edit-user.component.css']
})
export class EditUserComponent {
ulist:any=[];
eid:number=this.act.snapshot.params['id'];
constructor(public cs:CrudService,public act:ActivatedRoute,public rt:Router)
{

}
ngOnlit()
{
  this.fetchSingleUser(this.eid);
  
}

  fetchSingle(uid:number)
  {

    this.cs.getSingleUser(uid).subscribe(res=>
    {
      this.ulist=res;});
    }
    updateEmpolyee(uid:number,data:any)
    {
